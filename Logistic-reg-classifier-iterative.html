for j in range (0,iteracija):
    for i in range (0,len(X)):
        z[i]=w1*X[i,0]+w2*X[i,1]+b
        a[i]=sig(z[i])#racunanje procene
        J=J-(y[i]*np.log(a[i])+(1-y[i])*np.log(1-a[i]))#racunanje kost fje
        dz[i]=a[i]-y[i]
        dw1+=X[i,0]*dz[i]
        dw2+=X[i,1]*dz[i]
        db+=dz[i]
    dw1=dw1/m
    dw2=dw2/m
    db=db/m
    w1=w1-alf*dw1
    w2=w2-alf*dw2
    b=b-alf*db
